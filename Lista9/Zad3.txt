hyp <- 172
waga <-read.csv2("C:/Users/golde/OneDrive/Dokumenty/VPProjects/L-Statystyka-Dla-Inzynierow/Lista9/waga1.csv")

# Test Z
t <- (sqrt(length(sr)))*(mean(sr) - hyp)/(sd(sr))
t
p <- 2 * (1 - pnorm(abs(t)))
p

# Test Studenta
t <- (sqrt(length(sr)))*(mean(sr) - hyp)/(sd(sr))
t
p <- 2 * (1 - pt(abs(t), df=length(sr) - 1))
p

# Test t.test
t.test(sr, mu=hyp)
